proportion_fun <- function(x,cutoff){
  #统计x中大于cutoff的数量
  above_cutoff <- sum(x>cutoff,na.rm=TRUE)
  #统计x的总数
  total <- sum(!is.na(x))
  return (above_cutoff/total)#返回超过cutoff的比例
}
